    x:xs <- Just s
    return (x, xs)

stateOnMaybe :: String -> Maybe (Char, String)
stateOnMaybe = runStateT $ do
    x:xs <- get
    put xs
    return x

maybeOnState = runState $ runMaybeT $ do
    x:xs <- lift get
